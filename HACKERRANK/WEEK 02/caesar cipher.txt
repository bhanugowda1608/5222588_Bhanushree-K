char* caesarCipher(char* s, int k) {
int len = strlen(s);
    char* result = (char*)malloc((len + 1) * sizeof(char));
    
    k = k % 26;  
    for (int i = 0; i < len; i++) {
        char ch = s[i];
        
        if (isalpha(ch)) {
            char base = isupper(ch) ? 'A' : 'a';
            result[i] = (ch - base + k) % 26 + base;
        } else {
            result[i] = ch;
        }
    }
    result[len] = '\0';  
    return result;
}
